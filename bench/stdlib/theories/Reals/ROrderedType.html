Definition Reqb r1 r2 := if Req_dec r1 r2 then true else false.

forall r1 r2 : R, Reqb r1 r2 = true <-> r1 = r2


forall r1 r2 : R, Reqb r1 r2 = true <-> r1 = r2

 
r1, r2:R
NEQ:r1 <> r2
false = true <-> r1 = r2

 
r1, r2:R
NEQ:r1 <> r2
r1 = r2 -> false = true

Qed.

OrderedType structure for binary integers

Definition Rcompare x y :=
 match total_order_T x y with
  | inleft (left _) => Lt
  | inleft (right _) => Eq
  | inright _ => Gt
 end.
