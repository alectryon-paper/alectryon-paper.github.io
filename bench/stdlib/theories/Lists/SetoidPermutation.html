Context {A : Type} (eqA : relation A) (e : Equivalence eqA).

Inductive PermutationA : list A -> list A -> Prop :=
  | permA_nil: PermutationA nil nil
  | permA_skip x‚ÇÅ x‚ÇÇ l‚ÇÅ l‚ÇÇ :
     eqA x‚ÇÅ x‚ÇÇ -> PermutationA l‚ÇÅ l‚ÇÇ -> PermutationA (x‚ÇÅ :: l‚ÇÅ) (x‚ÇÇ :: l‚ÇÇ)
  | permA_swap x y l : PermutationA (y :: x :: l) (x :: y :: l)
  | permA_trans l‚ÇÅ l‚ÇÇ l‚ÇÉ :
     PermutationA l‚ÇÅ l‚ÇÇ -> PermutationA l‚ÇÇ l‚ÇÉ -> PermutationA l‚ÇÅ l‚ÇÉ.
