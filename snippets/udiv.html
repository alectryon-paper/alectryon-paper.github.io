= #include "stdint.h"

typedef struct {
  uint32_t quot;
  uint32_t rem;
} udiv_t;

udiv_t udiv(uint32_t num, uint32_t denom) {
  uint32_t q = 0;
  while (num >= denom) {
    num -= denom; ++q;
  }
  return (udiv_t) { .rem = num, .quot = q };
}: string
